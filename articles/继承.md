# 继承

## 原型链继承

    function Person(name, age) {
      this.name = name;
      this.age = age;
    }

    Person.prototype.getName = function () {
      return this.name;
    }

    function Child() {}

    // 实现原型链继承
    Child.prototype = new Person();

    let child = new Child();
    child.name = 'childA';
    child.getName();
