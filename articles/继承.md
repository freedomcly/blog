# 继承

## 原型链继承

    function Person(name, age) {
      this.name = name;
      this.age = age;
    }

    Person.prototype.getName = function () {
      return this.name;
    }

    function Child() {}
    function LittleChild() {}

    // 实现原型链继承
    Child.prototype = new Person();
    LittleChild.prototype = new Child();

    let child = new Child();
    let littleChild = new LittleChild();
    child.name = 'child';
    littleChild.name = 'little child';
    child.getName();
    littleChild.getName();
